    public int countPairs(int[] deliciousness) {
        int max=0,res=0;
        int mod=1000000007;

        //æ‰¾åˆ°æœ€å¤§çš„å€¼
        for(int i=0;i<deliciousness.length;++i){
            max=max<deliciousness[i]?deliciousness[i]:max;
        }
        Map<Integer,Integer> map=new HashMap<>();
        for(int i=0;i<deliciousness.length;++i){
            //æ¯æ¬¡åˆ¤æ–­å½“å‰è¿™ä¸ª2çš„å¹‚æ˜¯å¦è¢«å¯ä»¥ç»„æˆ

        }
        return res;
    }
}
