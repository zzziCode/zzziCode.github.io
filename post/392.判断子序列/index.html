class Solution {
public:
    bool isSubsequence(string s, string t) {
        vector<vector<int>>dp(s.size()+1,
                            vector<int>(t.size()+1,0));
        int res=0;
        for(int i=0;i<s.size();++i){
            for(int j=0;j<t.size();++j){
                if(s[i]==t[j]){
                    dp[i+1][j+1]=dp[i][j]+1;
                }else{
                    dp[i+1][j+1]=max(dp[i+1][j],dp[i][j+1]);
                }
                res=max(res,dp[i+1][j+1]);
            }
        }
        return res==s.size();
    }
};
