class Solution {
public:
    int climbStairs(int n) {
        if(n==1){
            return 1;
        }
        int left=1;
        int right=2;
        for(int i=3;i<=n;++i){
            int sum=right+left;
            left=right;
            right=sum;
        }
        return right;
    }
};
